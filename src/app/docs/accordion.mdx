## Usage

```tsx
import { Accordion } from '@ui';

export default () => (
  <>
    <Accordion>
      <Accordion.Item>
        <Accordion.Trigger />
        <Accordion.Content />
      </Accordion.Item>
    </Accordion>
  </>
);
```
